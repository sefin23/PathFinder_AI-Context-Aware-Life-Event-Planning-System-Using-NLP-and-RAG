Pythonapi/guides.py — the guide endpoint
@router.get('/tasks/{task_id}/guide')
async def get_task_guide(
    task_id: str,
    user: User = Depends(get_current_user)
):
    # 1. Get the task's guide_type
    task = db.get('tasks', task_id)
    if not task['guide_type']:
        raise HTTPException(404, 'No guide for this task')

    # 2. Get the guide
    guide = db.query(
        "SELECT * FROM task_guides WHERE task_type = %s",
        task['guide_type']
    )
    if not guide:
        raise HTTPException(404, 'No guide for this task')

    # 3. Prefill fields from vault + profile
    prefill = {}
    for key in (guide['prefill_keys'] or []):
        # Try vault first, then profile
        val = get_vault_field(user.id, key) or get_profile_field(user.id, key)
        if val:
            prefill[key] = {'value': val['value'], 'source': val['source']}

    return {
        'title': guide['title'],
        'est_minutes': guide['est_minutes'],
        'website': guide['website'],
        'steps': guide['steps'],
        'prefill': prefill,
        'result_items': guide['result_items']
    }

Pythonapi/vault_intelligence.py — Gemini 2.0 Flash document reading
import json, base64
from google import genai
from google.genai import types

# Same client you already use for plan generation
client = genai.Client()
MODEL = "gemini-2.0-flash"


async def extract_document_intelligence(
    vault_doc_id: str, user_id: str, plan_id: str
):
    doc = db.get('vault_documents', vault_doc_id)

    # Download the file from S3/storage
    doc_bytes = download_from_s3(doc['storage_url'])

    # Detect MIME type — Gemini handles PDF, JPG, PNG natively
    mime = get_mime_type(doc['storage_url'])  # 'application/pdf' | 'image/jpeg' | 'image/png'

    # Get plan context + user profile for mismatch detection
    plan = db.get('plans', plan_id)
    user_profile = db.get('user_profiles', user_id)

    prompt = f"""
You are reading a {doc['doc_type']} document for a user whose plan is:
{plan['description']}

Current user profile for mismatch checks:
{json.dumps(user_profile, indent=2)}

Extract the following fields. Return null for any field not found in the document.

FIELDS TO EXTRACT:
- full_name
- date_of_birth (ISO format YYYY-MM-DD)
- address (complete address string)
- aadhaar_number (mask middle 8 digits: XXXX XXXX 4821 format)
- pan_number
- joining_date (ISO format, for offer letters only)
- office_address (for offer letters only)
- employer_name (for offer letters only)
- ctc_annual (numeric, INR, for offer letters only)
- passport_expiry (ISO format, for passports only)

FLAGS TO CHECK:
- address_mismatch: if address in this document differs from profile address, describe the difference
- passport_urgency: if passport expires within 12 months, set to true
- employer_bgv_note: any BGV or background check requirements mentioned
- any_other_flags: array of any other actionable findings

Return ONLY a valid JSON object. No markdown, no explanation, no backticks.
Example shape:
{{
  "full_name": "Sefin K R",
  "joining_date": "2026-04-01",
  "office_address": "Infosys, Koramangala, Bangalore",
  "address_mismatch": "Document shows Mumbai address, profile shows Tirur Kerala",
  "passport_urgency": false,
  "employer_bgv_note": "Requires 3 years employment history",
  "any_other_flags": []
}}
"""

    # Send document + prompt to Gemini 2.0 Flash
    response = client.models.generate_content(
        model=MODEL,
        contents=[
            types.Part.from_bytes(
                data=doc_bytes,
                mime_type=mime,
            ),
            prompt,
        ],
        config=types.GenerateContentConfig(
            temperature=0.0,     # deterministic for extraction
            max_output_tokens=2000,
        )
    )

    # Parse the JSON response
    raw = response.text.strip()
    try:
        extracted = json.loads(raw)
    except json.JSONDecodeError:
        # Gemini occasionally wraps in ```json — strip it
        clean = raw.replace('```json', '').replace('```', '').strip()
        extracted = json.loads(clean)

    # Persist extracted fields
    db.insert('vault_extractions', {
        'vault_doc_id': vault_doc_id,
        'user_id': user_id,
        'extracted_fields': json.dumps(extracted),
        'flags': json.dumps(extracted.get('any_other_flags', []))
    })

    # Auto-apply to plan and collect action cards for UI
    actions = apply_to_plan(extracted, plan_id, user_id)

    return {
        'extracted': extracted,
        'actions_taken': actions,   # rendered as action cards in vault intel UI
        'flags': [
            extracted.get('address_mismatch'),
            extracted.get('employer_bgv_note'),
        ]
    }

def get_mime_type(url: str) -> str:
    """Infer MIME type from file extension in the storage URL."""
    url_lower = url.lower()
    if url_lower.endswith('.pdf'):  return 'application/pdf'
    if url_lower.endswith('.png'):  return 'image/png'
    if url_lower.endswith(('.jpg', '.jpeg')):  return 'image/jpeg'
    return 'application/pdf'  # safe default

        'user_id': user.id, 'name': file.filename,
        'doc_type': infer_doc_type(file.filename),
        'storage_url': url, 'size_bytes': len(doc_bytes)
    })

    # 3. Run Gemini extraction in BACKGROUND — don't block the response
    background_tasks.add_task(
        extract_document_intelligence,
        vault_doc_id=doc_id,
        user_id=user.id,
        plan_id=plan_id
    )

    # 4. Return immediately — frontend polls /vault/:id/intelligence
    return {
        'id': doc_id,
        'status': 'uploaded',
        'intelligence_status': 'processing'  # frontend shows spinner
    }


@router.get('/vault/{doc_id}/intelligence')
async def get_doc_intelligence(doc_id: str, user=Depends(get_current_user)):
    # Frontend polls this after upload to get the extracted actions
    extraction = db.query(
        "SELECT * FROM vault_extractions WHERE vault_doc_id = %s AND user_id = %s",
        doc_id, user.id
    )
    if not extraction:
        return {'status': 'processing'}  # still running

    return {
        'status': 'complete',
        'extracted': json.loads(extraction['extracted_fields']),
        'actions_taken': json.loads(extraction.get('actions_taken', '[]'))
    }

// After upload completes, poll until intelligence is ready
async function pollIntelligence(docId, onReady) {
  const poll = async () => {
    const res = await fetch(`/api/vault/${docId}/intelligence`)
    const data = await res.json()

    if (data.status === 'complete') {
      onReady(data)  // render the action cards
    } else {
      setTimeout(poll, 2000)  // retry every 2s
    }
  }
  poll()
}

Key points for Antigravity: (1) Use types.Part.from_bytes(data=doc_bytes, mime_type=mime) — Gemini 2.0 Flash reads PDFs and images natively, no conversion needed. (2) Set temperature=0.0 for extraction — you want deterministic JSON, not creative output. (3) Always run as a background_task — Gemini takes 2–5 seconds on a PDF and you never want to block the upload response.
